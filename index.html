// ====== Google Apps Script ======
// ใช้ Script เดียวจัดการทั้ง Sheets และ Drive

function doPost(e) {
  const data = JSON.parse(e.postData.contents);

  // --- บันทึกลง Google Sheets ---
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('บันทึกการส่ง');
  if (!sheet) {
    sheet = ss.insertSheet('บันทึกการส่ง');
    const headers = ['วันที่ส่ง','เวลาส่ง','ผู้ส่ง','ฝ่ายผู้ส่ง',
      'จากสาขา','รายการ','ลิงก์รูป','ผู้รับ','ฝ่ายผู้รับ',
      'ถึงสาขา','ผู้รับฝาก','ID'];
    sheet.appendRow(headers);
    sheet.getRange(1,1,1,headers.length)
      .setFontWeight('bold')
      .setBackground('#cc0000')
      .setFontColor('#ffffff');
  }

  // --- อัปโหลดรูปไป Google Drive ---
  let photoLinks = '';
  if (data.photos && data.photos.length > 0) {
    const folder = getOrCreateFolder('KY-Hub-Photos');
    const links = [];
    data.photos.forEach((b64, i) => {
      try {
        const parts = b64.split(',');
        const mime = parts[0].match(/:(.*?);/)[1];
        const ext = mime.split('/')[1];
        const blob = Utilities.newBlob(
          Utilities.base64Decode(parts[1]), mime,
          `${data.id}_${i+1}.${ext}`
        );
        const file = folder.createFile(blob);
        file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
        links.push(file.getUrl());
      } catch(err) { links.push('error'); }
    });
    photoLinks = links.join('\n');
  }

  sheet.appendRow([
    data.date, data.time, data.sender, data.dept,
    data.from, data.items, photoLinks,
    data.receiver, data.receiverDept, data.to, data.custodian, data.id
  ]);

  return ContentService
    .createTextOutput(JSON.stringify({status:'ok', photos: photoLinks}))
    .setMimeType(ContentService.MimeType.JSON);
}

function getOrCreateFolder(name) {
  const folders = DriveApp.getFoldersByName(name);
  return folders.hasNext() ? folders.next() : DriveApp.createFolder(name);
}
